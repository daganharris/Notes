FUNCTION ComputeCansNeeded(firstWallArea : REAL, secondWallArea : REAL, excludedWallArea : REAL) RETURNS outputStat
    DECLARE returnStruct : REAL
    cansNeeded ← ((firstWallArea + secondWallArea) * 2 - excludedWallArea) / paintCoverage / canSize
    RETURN ROUND(cansNeeded)
END FUNCTION

FUNCTION GetExcludedAreas() RETURNS REAL
    DECLARE length : REAL
    DECLARE width : REAL
    DECLARE totalArea : REAL
    
    IF BoolInput("Do you have any areas to exclude (windows, doors, outlets etc.)", FALSE) THEN
        length ← ValInput("Enter the exclusion area's length.", FALSE)
        width ← ValInput("Enter the exclusion area's length.", FALSE)
        totalArea ← length * width

        DECLARE moreExclusions : BOOLEAN
        moreExclusions ← TRUE
        WHILE moreExclusions = TRUE
            IF NOT BoolInput("Would you like to add another exclusion?", FALSE) THEN
                length ← ValInput("Enter another exclusion area's length.", FALSE)
                width ← ValInput("Enter another exclusion area's width.", FALSE)
                totalArea ← totalArea + (length * width)
            ELSE 
                moreExclusions ← FALSE
            ENDIF
        ENDWHILE
        RETURN totalArea
    ENDIF
    RETURN 0;
ENDFUNCTION

FUNCTION BoolInput(message : STRING, defaultChoice : BOOLEAN) RETURNS BOOLEAN
    DECLARE choiceHelper : STRING 
    IF defaultChoice = TRUE THEN
        choiceHelper ← "(Y/n)"
    ELSE
        choiceHelper ← "(y/N)"
    ENDIF

    OUTPUT(message + choiceHelper)
    DECLARE inputChoice : STRING
    INPUT inputChoice
    CASE OF inputChoice
        "Y": RETURN TRUE
        "y": RETURN TRUE
        "N": RETURN FALSE
        "n": RETURN FALSE
        OTHERWISE: RETURN FALSE
    ENDCASE
ENDFUNCTION

FUNCTION GetWallArea(isOther : BOOLEAN) RETURNS REAL
    DECLARE length : REAL
    DECALRE width : REAL
    DECLARE outputTemplate : STRING
    IF isOther = TRUE THEN
        outputTemplate ← "Please enter the shorter wall"
    ELSE
        outputTemplate ← "Please enter the longer wall"
    ENDIF

    length ← ValInput(outputTemplate + "'s length in metres", TRUE)
    width ← ValInput(outputTemplate + "'s length in metres", TRUE)

    RETURN length * width
ENDFUNCTION

FUNCTION ValInput(message : STRING, allowingNegative : BOOLEAN) RETURN BOOLEAN
    OUTPUT(message)
    DECLARE inputVal : REAL
    WHILE TRUE
        INPUT inputVal
        IF inputVal < 0  OR allowingNegative = TRUE THEN
            RETURN inputVal
        ELSE
            OUTPUT("ERROR: invalid input value, try again.")
        ENDIF
    ENDWHILE
ENDFUNCTION

CONSTANT paintCoverage ← 6.25 # m^2 / l
CONSTANT canSize ← 5 # litres
CONSTANT canPrice ← 15.5

DECLARE firstWallArea : REAL
DECLARE secondWallArea : REAL
DECLARE excludedWallArea : REAL

DECLARE cansNeeded : INTEGER
DECLARE grossCost : REAL

cansNeeded ← ComputePaintStats(firstWallArea, secondWallArea, excludedWallArea)
grossCost ← cansNeeded * canPrice

OUTPUT("You need " + cansNeeded + " cans to complete this job.")
OUTPUT("This will cost £" + grossCost + " to purchase.")
